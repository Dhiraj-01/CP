### Problem
```
Given a string of N length consisting of only lower case latin characters.
Find lexicographically smallest subsequence of length exactly equal to X of this string.

constraints:
1 <= N <= 1e6
1 <= X <= N
'a' <= s[i] <= 'z'

input:
N = 7, X = 5 
String = "tourist"

Output: 
orist

Among all other subsequence of length x = 5, this "orist" subsequence is the smallest of them all.
```

// return lexicographilly smallest subsequence of string s with len
string f(string s, ll len)
{
    ll n = s.size();
    vector<vector<ll>> pos(n, vector<ll>(26, n));
    pos[n - 1][s[n - 1] - 'a'] = n - 1;
    for(ll i = n - 2; i >= 0; i--) {
        for(ll j = 0; j < 26; j++) {
            pos[i][j] = pos[i + 1][j];
        }
        pos[i][s[i] - 'a'] = i;
    }

    ll last_i = 0;
    string ans = "";
    for(ll i = len; i >= 1; i--)
    {
        bool ok = 0;
        for(ll j = 0; j < 26; j++)
        {
            ll x = n - pos[last_i][j];
            if(x >= i) {
                ok = 1;
                ans += ('a' + j);
                last_i = pos[last_i][j] + 1;
                break;
            }
        }
        if(!ok) return "-1";
    }
    return ans;
}

void solve(ll &tc)
{
    ll x;
    string s;
    cin >> s >> x;
    cout << f(s, x) << endl;
}
